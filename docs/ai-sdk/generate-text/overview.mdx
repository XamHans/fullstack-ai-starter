## Basic Usage

<Tabs>
  <Tab title="Simple Generation">
    Generate text with a straightforward prompt:

    ```typescript
    const response = await fetch('/api/generate-text', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${userToken}` // Authentication required
      },
      body: JSON.stringify({
        prompt: 'Write a professional email about project updates'
      })
    });

    const { text } = await response.json();
    console.log(text);
    ```
  </Tab>

  <Tab title="React Hook">
    Use our custom hook for React components:

    ```typescript
    import { useGenerateText } from '@/hooks/use-generate-text';

    function TextGenerator() {
      const { generateText, loading, error } = useGenerateText();

      const handleGenerate = async () => {
        const result = await generateText({
          prompt: 'Create a product description for a smart watch'
        });

        if (result) {
          console.log('Generated:', result.text);
        }
      };

      return (
        <button
          onClick={handleGenerate}
          disabled={loading}
        >
          {loading ? 'Generating...' : 'Generate Text'}
        </button>
      );
    }
    ```
  </Tab>

  <Tab title="Server-Side">
    Generate text in API routes or server components:

    ```typescript
    import { generateText } from 'ai';
    import { openai } from '@ai-sdk/openai';

    export async function generateProductDescription(productName: string) {
      const { text } = await generateText({
        model: openai('gpt-4'),
        prompt: `Create a compelling product description for: ${productName}`,
        maxTokens: 300,
      });

      return text;
    }
    ```
  </Tab>
</Tabs>

<Accordion title="Custom System Instructions">
  Provide context and personality to your AI:

  ```typescript
  const { text } = await generateText({
    model: openai('gpt-4'),
    messages: [
      {
        role: 'system',
        content: 'You are a professional copywriter specializing in tech products. Write in a friendly but authoritative tone.'
      },
      {
        role: 'user',
        content: 'Create a landing page headline for our new productivity app'
      }
    ]
  });
  ```
</Accordion>

<Accordion title="Temperature & Creativity Control">
  Fine-tune the randomness and creativity of outputs:

  ```typescript
  // More creative and varied (0.0 - 1.0)
  const creativeText = await generateText({
    model: openai('gpt-4'),
    prompt: 'Write a creative story',
    temperature: 0.9, // High creativity
    topP: 0.9
  });

  // More consistent and factual
  const factualText = await generateText({
    model: openai('gpt-4'),
    prompt: 'Explain quantum computing',
    temperature: 0.2, // Low creativity
    topP: 0.8
  });
  ```
</Accordion>

<Accordion title="Token Limits & Chunking">
  Handle long content generation efficiently:

  ```typescript
  async function generateLongContent(sections: string[]) {
    const results = [];

    for (const section of sections) {
      const { text } = await generateText({
        model: openai('gpt-4'),
        prompt: `Write detailed content about: ${section}`,
        maxTokens: 1000, // Limit per section
      });

      results.push(text);
    }

    return results.join('\n\n');
  }
  ```
</Accordion>

## Performance Optimization

<Tabs>
  <Tab title="Caching Strategies">
    Cache frequently requested content:

    ```typescript
    import { unstable_cache } from 'next/cache';

    const generateCachedText = unstable_cache(
      async (prompt: string) => {
        const { text } = await generateText({
          model: openai('gpt-3.5-turbo'),
          prompt
        });
        return text;
      },
      ['generate-text'],
      { revalidate: 3600 } // Cache for 1 hour
    );
    ```
  </Tab>

  <Tab title="Request Batching">
    Process multiple requests efficiently:

    ```typescript
    async function batchGenerate(prompts: string[]) {
      const promises = prompts.map(prompt =>
        generateText({
          model: openai('gpt-3.5-turbo'),
          prompt,
          maxTokens: 200
        })
      );

      const results = await Promise.all(promises);
      return results.map(r => r.text);
    }
    ```
  </Tab>

  <Tab title="Streaming for UX">
    Provide immediate feedback with streaming:

    ```typescript
    import { streamText } from 'ai';

    const result = streamText({
      model: openai('gpt-4'),
      prompt: 'Write a detailed article about AI'
    });

    for await (const chunk of result.textStream) {
      // Update UI progressively
      setDisplayText(prev => prev + chunk);
    }
    ```
  </Tab>
</Tabs>

## Error Handling & Debugging

<Warning>
Always implement proper error handling for production applications:
</Warning>

```typescript
async function safeGenerateText(prompt: string) {
  try {
    const response = await fetch('/api/generate-text', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ prompt })
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.message || 'Generation failed');
    }

    return await response.json();
  } catch (error) {
    console.error('Text generation error:', error);

    // Provide fallback or user-friendly message
    return {
      text: 'Sorry, I couldn\'t generate content right now. Please try again.',
      error: true
    };
  }
}
```
